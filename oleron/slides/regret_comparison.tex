\begin{figure}[H]
\centering
\resizebox{0.92\textwidth}{!}{%
\begin{tikzpicture}
\pgfmathsetmacro{\K}{10}

\begin{semilogyaxis}[
    width=\textwidth,
    height=6.8cm,
    samples=250,
    domain=10:1e6,
    xmin=10, xmax=1e6,
    xlabel={Time Horizon $T$},
    ylabel={Expected Regret $R(T)$},
    legend style={
        at={(0.97,0.03)},
        anchor=south east,
        font=\small,
        fill=white,
        draw=none
    },
    grid=major,
    log ticks with fixed point,
    tick label style={font=\small},
    label style={font=\small},
]

    % --- UCB: O(sqrt(K T log T)) ---
    \addplot[ultra thick, blue] {sqrt(\K*x*ln(x))};
    \addlegendentry{UCB}

    % --- ETC (untuned): ~O(K^{1/3} T^{2/3}) ---
    \addplot[ultra thick, dashed, red] {(\K)^(1/3)*(x)^(2/3)*(ln(x))^(1/3)};
    \addlegendentry{ETC}

    % --- AE (fixed budget): O(T) ---
    \addplot[ultra thick, dotted, teal] {0.025*x};
    \addlegendentry{AE}

\end{semilogyaxis}
\end{tikzpicture}%
}
\end{figure}
